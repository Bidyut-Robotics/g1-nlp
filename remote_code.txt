import sys
import threading
import time

from unitree_sdk2py.core.channel import ChannelFactoryInitialize, ChannelSubscriber
from unitree_sdk2py.g1.audio.g1_audio_client import AudioClient
from unitree_sdk2py.idl.unitree_hg.msg.dds_ import LowState_
from wav import amplify_pcm, pcm_duration_sec, play_pcm_stream, prepare_pcm_for_g1_playstream, read_wav


DEFAULT_LOWSTATE_TOPIC = "rt/lf/lowstate"
APP_NAME = "remote_f3_audio"
G1_PCM_RATE = 16000
MAX_VOLUME = 100
# ~200 ms per chunk at 16 kHz mono (see Unitree g1-audio-driver PLAYSTREAM_CHUNK_BYTES)
PLAY_CHUNK_BYTES = 6400


def _parse_argv_with_boost(argv_tail):
    """Strip --boost N from argv tail; return (remaining_args, boost)."""
    boost = 1.0
    out = []
    i = 0
    while i < len(argv_tail):
        if argv_tail[i] == "--boost" and i + 1 < len(argv_tail):
            try:
                boost = max(1.0, min(5.0, float(argv_tail[i + 1])))
            except ValueError:
                print(f"[ERROR] Invalid --boost value: {argv_tail[i + 1]}")
                sys.exit(1)
            i += 2
            continue
        out.append(argv_tail[i])
        i += 1
    return out, boost


class F3AudioTrigger:
    def __init__(self, audio_client, pcm_list):
        self.audio_client = audio_client
        self.pcm_list = pcm_list
        self.previous_f3 = 0
        self.is_playing = False
        self.lock = threading.Lock()

    def low_state_handler(self, msg):
        wireless_remote = msg.wireless_remote
        if len(wireless_remote) < 4:
            return

        f3_pressed = (wireless_remote[2] >> 7) & 1
        if f3_pressed and not self.previous_f3:
            self.play_once()
        self.previous_f3 = f3_pressed

    def play_once(self):
        with self.lock:
            if self.is_playing:
                print("[INFO] F3 pressed, audio is already playing")
                return
            self.is_playing = True

        print("[INFO] F3 pressed, playing audio")
        thread = threading.Thread(target=self._play_audio, daemon=True)
        thread.start()

    def _play_audio(self):
        try:
            duration = pcm_duration_sec(self.pcm_list, G1_PCM_RATE, num_channels=1)
            print(f"[INFO] PCM duration (16 kHz mono): {duration:.2f} s")
            t0 = time.time()
            play_pcm_stream(
                self.audio_client,
                self.pcm_list,
                APP_NAME,
                chunk_size=PLAY_CHUNK_BYTES,
                sample_hz=G1_PCM_RATE,
            )
            elapsed = time.time() - t0
            # Let the last samples drain from the voice pipeline before PlayStop.
            tail = max(0.35, min(6.0, duration - elapsed + 0.45))
            time.sleep(tail)
            self.audio_client.PlayStop(APP_NAME)
            print("[INFO] Audio playback finished")
        finally:
            with self.lock:
                self.is_playing = False


def main():
    if len(sys.argv) >= 2 and sys.argv[1] == "--check":
        if len(sys.argv) < 3:
            print(f"Usage: python3 {sys.argv[0]} --check <wav_file_path>")
            sys.exit(1)
        wav_path = sys.argv[2]
        pcm_list, sample_rate, num_channels, is_ok = read_wav(wav_path)
        if not is_ok:
            print(f"[FAIL] {wav_path}: read_wav failed")
            sys.exit(1)
        try:
            prep = prepare_pcm_for_g1_playstream(pcm_list, sample_rate, num_channels)
        except ValueError as e:
            print(f"[FAIL] {wav_path}: {e}")
            sys.exit(1)
        dur_src = pcm_duration_sec(pcm_list, sample_rate, num_channels=num_channels)
        dur_g1 = pcm_duration_sec(prep, G1_PCM_RATE, num_channels=1)
        print(f"[OK] {wav_path}")
        print(f"     source: {sample_rate} Hz, {num_channels} ch, {dur_src:.2f} s, {len(pcm_list)} bytes PCM")
        print(f"     G1 prep: {G1_PCM_RATE} Hz mono, {dur_g1:.2f} s, {len(prep)} bytes PCM")
        sys.exit(0)

    if len(sys.argv) < 3:
        print(
            f"Usage: python3 {sys.argv[0]} <network_interface> <wav_file_path> "
            f"[lowstate_topic] [--boost <1.0-5.0>]"
        )
        print(f"       python3 {sys.argv[0]} --check <wav_file_path>")
        print("  --boost: optional software gain after 16 kHz prep (default 1.0)")
        print(f"Default lowstate topic: {DEFAULT_LOWSTATE_TOPIC}")
        sys.exit(1)

    filtered, boost = _parse_argv_with_boost(sys.argv[1:])
    if len(filtered) < 2:
        print(
            f"Usage: python3 {sys.argv[0]} <network_interface> <wav_file_path> "
            f"[lowstate_topic] [--boost <1.0-5.0>]"
        )
        sys.exit(1)

    network_interface = filtered[0]
    wav_path = filtered[1]
    lowstate_topic = filtered[2] if len(filtered) > 2 else DEFAULT_LOWSTATE_TOPIC

    pcm_list, sample_rate, num_channels, is_ok = read_wav(wav_path)
    print(f"[DEBUG] Read success: {is_ok}")
    print(f"[DEBUG] Sample rate: {sample_rate} Hz")
    print(f"[DEBUG] Channels: {num_channels}")
    print(f"[DEBUG] PCM byte length: {len(pcm_list)}")

    if not is_ok or sample_rate <= 0 or num_channels < 1:
        print("[ERROR] Failed to read WAV or invalid format (need 16-bit PCM)")
        sys.exit(1)

    try:
        pcm_list = prepare_pcm_for_g1_playstream(pcm_list, sample_rate, num_channels)
    except ValueError as e:
        print(f"[ERROR] {e}")
        sys.exit(1)

    if boost > 1.0:
        print(f"[INFO] Software boost: {boost}x")
        pcm_list = amplify_pcm(pcm_list, boost)

    ChannelFactoryInitialize(0, network_interface)

    audio_client = AudioClient()
    audio_client.SetTimeout(10.0)
    audio_client.Init()
    try:
        ret = audio_client.SetVolume(MAX_VOLUME)
        print(f"[INFO] SetVolume({MAX_VOLUME}) ret={ret}")
    except Exception as e:
        print(f"[WARN] SetVolume failed: {e}")

    trigger = F3AudioTrigger(audio_client, pcm_list)
    lowstate_subscriber = ChannelSubscriber(lowstate_topic, LowState_)
    lowstate_subscriber.Init(trigger.low_state_handler, 10)

    print(f"[INFO] Listening on {lowstate_topic}")
    print(f"[INFO] Boost={boost}x (hardware volume {MAX_VOLUME}, software 1.0–5.0 via --boost)")
    print("[INFO] Press F3 on the remote controller to play the audio")

    try:
        while True:
            time.sleep(1)
    except KeyboardInterrupt:
        audio_client.PlayStop(APP_NAME)
        print("\n[INFO] Stopped")


if __name__ == "__main__":
    main()
